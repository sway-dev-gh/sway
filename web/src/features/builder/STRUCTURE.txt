HOOK APIS
=========

useCanvasHistory()
  Returns: { saveToHistory, undo, redo, canUndo, canRedo, clearHistory }

useCanvasElements(saveToHistory)
  Returns: { elements, selectedElement, lockedElements, clipboard,
             addElement, updateElement, updateElementProperty,
             deleteElement, duplicateElement, copyToClipboard,
             pasteFromClipboard, isElementLocked, toggleElementLock,
             bringForward, sendBackward, bringToFront, sendToBack }

useCanvasDrag({ snapToGrid, gridSize })
  Returns: { isDragging, draggedComponent, startComponentDrag,
             endComponentDrag, isDraggingElement, dragOffset,
             startElementDrag, endElementDrag, calculateDragPosition,
             isResizing, resizeHandle, startResize, endResize,
             calculateResizeSize, snapToGridValue, reset }

useCanvasZoom({ initialZoom, minZoom, maxZoom, step })
  Returns: { zoom, zoomPercentage, canZoomIn, canZoomOut,
             zoomIn, zoomOut, resetZoom, setZoomLevel }
